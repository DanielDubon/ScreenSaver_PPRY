# Screensaver paralelo optimizado con OpenMP avanzado
cmake_minimum_required(VERSION 3.16)
project(screensaver_omp_optimized LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

find_package(PkgConfig REQUIRED)
find_package(OpenMP REQUIRED)

# SDL2
pkg_check_modules(SDL2 REQUIRED sdl2)

add_executable(screensaver_omp_optimized base/src/main_optimized.cpp)

# Includes
target_include_directories(screensaver_omp_optimized PRIVATE
    ${SDL2_INCLUDE_DIRS}
)

# Librerias
target_link_libraries(screensaver_omp_optimized
    OpenMP::OpenMP_CXX
    ${SDL2_LIBRARIES}
)

# Flags extra
target_compile_options(screensaver_omp_optimized PRIVATE
    ${SDL2_CFLAGS_OTHER}
    -O3
    -march=native
)
